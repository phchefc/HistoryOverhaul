# Setting on the European religions before the Reformation
country_event = {
	id = eh_humanities.5
	title = "eh_humanities.5.t"
	desc = "eh_humanities.5.d"
	picture = RELIGIOUS_CONVERSION_eventPicture
	
	trigger = {
		OR = {
			AND = {
				tag = PAP
				is_year = 1440
				NOT = { is_year = 1450 }
				NOT = { has_country_flag = before_reformation }
			}
			AND = {
				tag = FRA
				is_year = 1470
				NOT = { is_year = 1480 }
				NOT = { has_country_flag = before_reformation }
			}
			AND = {
				tag = HAB
				is_year = 1500
				NOT = { is_year = 1510 }
				NOT = { has_country_flag = before_reformation }
			}
		}
		has_global_flag = eh_historic
	}
	
	mean_time_to_happen = {
		days = 700
	}

	immediate = {
		hidden_effect = {
			france_region = { change_religion = catholic }
			low_countries_region = { change_religion = catholic }
			italy_region = { change_religion = catholic }
			north_german_region = {
				limit = {
					AND = {
						NOT = { area = bohemia_area }
						NOT = { area = moravia_area }
						NOT = { area = erzgebirge_area }
					}
				}
				change_religion = catholic
			}
			bohemia_area = { change_religion = hussite }
			moravia_area = { change_religion = hussite }
			erzgebirge_area = { change_religion = hussite }
			south_german_region = { change_religion = catholic }
			british_isles_region = { change_religion = catholic }
			baltic_region = { change_religion = catholic }
			poland_region = { change_religion = catholic }
			croatia_area = { change_religion = catholic }
			iberia_region = {
				limit = {
					AND = {
						NOT = { area = lower_andalucia_area }
						NOT = { area = upper_andalucia_area }
					}
				}
				change_religion = catholic
			}
			scandinavia_region = {
				limit = {
					AND = {
						NOT = { area = laponia_area }
						NOT = { area = norrland_area }
						NOT = { area = bothnia_area }
						NOT = { area = northern_norway }
					}
				}
				change_religion = catholic
			}
			carpathia_region = { 
				limit = {
					AND = {
						NOT = { area = wallachia_area }
						NOT = { area = moldavia_area }
						NOT = { area = bessarabia_area }
					}
				}
				change_religion = catholic
			}
			carpathia_region = {
				limit = {
					AND = {
						area = wallachia_area
						area = moldavia_area
						area = bessarabia_area
					}
				}
				change_religion = orthodox
			}
			balkan_region = {
				limit = {
					AND = {
						NOT = { area = croatia_area }
						NOT = { area = east_adriatic_coast_area }
						NOT = { area = slavonia_area }
					}
				}
				change_religion = orthodox
			}
			russia_region = { change_religion = orthodox }
			ruthenia_region = { change_religion = orthodox }
			anatolia_region = {
				limit = { owned_by = BYZ }
				change_religion = orthodox
			}
		}
	}

	option = {
		name = "eh_humanities.5.a"
		ai_chance = { factor = 100 }
		ROOT = { set_country_flag = before_reformation }
		every_country = {
			limit = {
				OR = {
					capital_scope = { region = france_region }
					capital_scope = { region = scandinavia_region }
					capital_scope = { region = low_countries_region }
					capital_scope = { region = italy_region }
					capital_scope = { region = north_german_region }
					capital_scope = { region = south_german_region }
					capital_scope = { region = british_isles_region }
					capital_scope = { region = baltic_region }
					capital_scope = { region = poland_region }
					capital_scope = { area = croatia_area }
					capital_scope = {
						AND = {
							region = carpathia_region
							NOT = { area = wallachia_area }
							NOT = { area = moldavia_area }
							NOT = { area = bessarabia_area }
						}
					}
				}
			}
			change_religion = catholic
		}
		every_country = {
			limit = {
				OR = {
					capital_scope = {
						AND = {
							region = balkan_region
							NOT = { area = croatia_area }
							NOT = { owned_by = TUR }
						}
					}
					capital_scope = { region = russia_region }
					capital_scope = { region = ruthenia_region }
					capital_scope = { area = wallachia_area }
					capital_scope = { area = moldavia_area }
					capital_scope = { area = bessarabia_area }
				}
			}
			change_religion = orthodox
		}
	}
}
