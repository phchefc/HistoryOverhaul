# Religion in Scotland
province_event = {
	id = eh_britain.71
	title = "eh_britain.71.t"
	desc = "eh_britain.71.d"
	picture = RELIGIOUS_CONVERSION_eventPicture
	
	trigger = {
		OR = {
			area = highlands_area
			area = lowlands_area
			area = kingdom_of_the_isles_area
		}
		NOT = { religion = reformed }
		is_year = 1550
	}
	
	mean_time_to_happen = {
		months = 100

		modifier = {
			factor = 0.8
			is_year = 650
		}
		modifier = {
			factor = 0.8
			any_neighbor_province = { religion = anglican }
		}
	}
	
	option = {
		name = "eh_britain.71.a"
		ai_chance = { factor = 100 }
		every_province = {
			limit = {
				AND = {
					OR = {
						area = highlands_area
						area = lowlands_area
						area = kingdom_of_the_isles_area
					}
					NOT = { religion = reformed }
				}
			}
			change_religion = reformed
		}
	}
}
