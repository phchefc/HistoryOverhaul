# Bestow Imperial Grace - Chinese empires
country_event = {
	id = eh_diplomacy.1
	title = "eh_diplomacy.1.t"
	desc = "eh_diplomacy.1.d"
	picture = GIFTS_TO_EMPEROR_eventPicture

	trigger = {
		ROOT = { is_emperor_of_china = yes }
		is_year = 2
		NOT = { is_year = 1870 }
		NOT = { is_at_war = yes }
	}
	
	mean_time_to_happen = {
		days = 700
	}
	
	option = {
		name = "eh_diplomacy.1.a"
		ai_chance = { factor = 50 }
		ROOT = {
			add_treasury = -1000
			add_adm_power = -10
			add_dip_power = -10
			add_mil_power = -10
			add_mandate = 1
			add_meritocracy = 1
		}
		every_subject_country = {
			limit = {
				is_subject_of_type_with_overlord = { who = ROOT type = tributary_state }
			}
			country_event = { id = eh_diplomacy.2 days = 7 }
		}
	}
	
	option = {
		name = "eh_diplomacy.1.b"
		ai_chance = { factor = 50 }
		ROOT = {
			add_mandate = -1
			add_meritocracy = -1
		}
		every_subject_country = {
			limit = {
				is_subject_of = ROOT
			}
			add_liberty_desire = 1
		}
	}
}
